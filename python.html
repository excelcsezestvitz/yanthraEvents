def first(arr, low, high, x, n) :
if (high >= low) :
    mid = low + (high - low) // 2; 
    if ((mid == 0 or x > arr[mid-1]) and arr[mid] == x) :
        return mid
    if (x > arr[mid]) :
        return first(arr, (mid + 1), high, x, n)
    return first(arr, low, (mid -1), x, n)
                                     
return -1
